08 import demo.tasklist.common.Constants;
09 import demo.tasklist.service.DataKeeper;
10 import javax.servlet.http.HttpServletRequest;
11 import javax.servlet.http.HttpServletResponse;
12 import javax.servlet.http.HttpSession;

15 import org.apache.struts.action.ActionForward;
16 import org.apache.struts.action.ActionMapping;
17 
18 /**
19  * WelcomeAction initializes objects used by displayUserList.jsp 
20  */
21 public class WelcomeAction extends Action {
22    public ActionForward execute( ActionMapping mapping,
23                                  ActionForm form,
24                                  HttpServletRequest request,
25                                  HttpServletResponse response)
26     throws Exception {
27 
28       HttpSession session = request.getSession();
29       
30       if (session.getAttribute(Constants.DATA_KEY) == null) {
31         DataKeeper dkeeper = new DataKeeper();
32         session.setAttribute( Constants.DATA_KEY, dkeeper);
33       }
34       return mapping.findForward(Constants.SUCCESS_KEY );
35   }
36 }
